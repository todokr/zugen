@import zugen.core.document.DomainRelationDiagramDocument.Digraph
@import zugen.core.document.DomainRelationDiagramDocument.{DomainInternalEdge, DomainExternalEdge}

@(digraph: Digraph)
digraph domain_relation_diagram {
  graph [
    charset = "UTF-8";
    bgcolor = "#343434",
    fontcolor = "#eeeeee",
    style = "filled",
    rankdir = TB,
    margin = 0.2,
    nodesep = 0.5,
    ranksep = 0.8,
    compound = true,
    layout = dot,
    pad = 0.1
  ];

  node [
    colorscheme = rdylgn11,
    fontname = "sans-serif",
    color = 7,
    fontsize = 10,
    fontcolor = 6,
    fillcolor = 11,
    margin = 0.05,
    style = "solid,filled"
  ];

  edge [
    style = solid,
    fontsize = 10,
    color = "#cccccc",
    labelfloat = false
  ];

  @for(subGraph <- digraph.subGraphs) {
    subgraph cluster_@{subGraph.id} {
      label = "  @subGraph.label  ";
      style = "";
      color = "#eeeeee";
      fontname = "sans-serif";
      fontsize = 10;
      margin = 1;
      pad = 1;

      @for(node <- subGraph.nodes) {
        @node.id [label = "@node.label"]
      }
    }
  }

  @for(edge <- digraph.edges) {
    @edge match {
      case DomainInternalEdge(from, to, arrowHead) => { @from -> @to [arrowhead = @arrowHead] }
      case DomainExternalEdge(from, toLabel, _, arrowHead) => { @from -> "@toLabel" [ color = "#880000", arrowhead = @arrowHead] }
    }
  }
}
